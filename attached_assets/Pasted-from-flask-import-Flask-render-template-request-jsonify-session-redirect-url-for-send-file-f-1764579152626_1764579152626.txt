@app.after_request
def add_header(response):
    if response.content_type and 'text/html' in response.content_type:
        response.headers['Cache-Control'] = 'no-store, no-cache, must-revalidate, max-age=0'
        response.headers['Pragma'] = 'no-cache'
        response.headers['Expires'] = '0'
    return response

# Auth Decorators
def login_required(f):
    @wraps(f)
    def decorated_function(*args, **kwargs):
        if 'user_id' not in session:
            return redirect(url_for('login_page'))
        return f(*args, **kwargs)
    return decorated_function

def admin_required(f):
    @wraps(f)
    def decorated_function(*args, **kwargs):
        if 'user_id' not in session:
            return redirect(url_for('login_page'))
        if session.get('role') not in ['Super Admin', 'Assistant Admin']:
            return jsonify({'error': 'Unauthorized'}), 403
        return f(*args, **kwargs)
    return decorated_function

@app.route('/api/login', methods=['POST'])
def login():
    data = request.json
    username = data.get('username')
    password = data.get('password')
    
    conn = get_db()
    cursor = conn.cursor(cursor_factory=psycopg2.extras.RealDictCursor)
    try:
        cursor.execute('''
            SELECT u.*, r.name as role, rt.name as reseller_tier, b.name as managed_brand_name
            FROM users u
            JOIN roles r ON u.role_id = r.id
            LEFT JOIN reseller_tiers rt ON u.reseller_tier_id = rt.id
            LEFT JOIN brands b ON u.managed_brand_id = b.id
            WHERE u.username = %s
        ''', (username,))
        user = cursor.fetchone()
        
        if not user or not bcrypt.checkpw(password.encode('utf-8'), user['password'].encode('utf-8')):
            return jsonify({'error': 'Invalid credentials'}), 401
        
        session['user_id'] = user['id']
        session['username'] = user['username']
        session['full_name'] = user['full_name']
        session['role'] = user['role']
        session['reseller_tier'] = user['reseller_tier']
        session['managed_brand_id'] = user['managed_brand_id'] # Store managed brand in session
        
        redirect_url = '/admin' if user['role'] in ['Super Admin', 'Assistant Admin'] else '/dashboard'
        
        return jsonify({
            'message': 'Login successful',
            'redirect': redirect_url,
            'user': {
                'full_name': user['full_name'],
                'role': user['role'],
                'managed_brand_name': user['managed_brand_name']
            }
        })
    finally:
        cursor.close()
        conn.close()

@app.route('/api/brands', methods=['POST'])
@admin_required
def create_brand():
    # Only Super Admin can create brands
    if session.get('role') != 'Super Admin':
        return jsonify({'error': 'Only Super Admin can create brands'}), 403

    data = request.json
    brand_name = data.get('name')
    if not brand_name:
        return jsonify({'error': 'Brand name is required'}), 400
        
    conn = get_db()
    cursor = conn.cursor()
    try:
        cursor.execute('INSERT INTO brands (name) VALUES (%s) RETURNING id, name', (brand_name,))
        new_brand = cursor.fetchone()
        conn.commit()
        return jsonify({'id': new_brand[0], 'name': new_brand[1]}), 201
    except Exception as e:
        conn.rollback()
        return jsonify({'error': str(e)}), 500
    finally:
        conn.close()

@app.route('/api/products', methods=['GET'])
@admin_required
def get_products():
    role = session.get('role')
    managed_brand_id = session.get('managed_brand_id')
    
    conn = get_db()
    cursor = conn.cursor(cursor_factory=psycopg2.extras.RealDictCursor)
    
    query = '''
        SELECT p.*, b.name as brand_name,
        (SELECT COUNT(DISTINCT id) FROM skus WHERE product_id = p.id) as sku_count,
        (SELECT image_url FROM product_images WHERE product_id = p.id ORDER BY sort_order LIMIT 1) as first_image_url
        FROM products p
        LEFT JOIN brands b ON p.brand_id = b.id
    '''
    params = []
    
    # FILTER: If Assistant Admin, show only their brand
    if role == 'Assistant Admin' and managed_brand_id:
        query += ' WHERE p.brand_id = %s'
        params.append(managed_brand_id)
        
    query += ' ORDER BY p.created_at DESC'
    
    cursor.execute(query, tuple(params))
    products = [dict(r) for r in cursor.fetchall()]
    conn.close()
    return jsonify(products)
