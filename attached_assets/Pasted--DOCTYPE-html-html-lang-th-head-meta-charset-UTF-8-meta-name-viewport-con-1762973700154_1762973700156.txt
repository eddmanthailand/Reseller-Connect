    <script>
        document.addEventListener('DOMContentLoaded', () => {
            const optionsContainer = document.getElementById('options-container');
            const addOptionBtn = document.getElementById('add-option-btn');
            const generateVariantsBtn = document.getElementById('generate-variants-btn');
            const variantsSection = document.getElementById('variants-section');
            const variantsTableContainer = document.getElementById('variants-table-container');
            const optionTemplate = document.getElementById('option-block-template');

            // --- ส่วนที่ 2: การจัดการ Options และ Values ---

            // 2.1 เพิ่ม Option Block ใหม่
            addOptionBtn.addEventListener('click', () => {
                const newOptionBlock = optionTemplate.content.cloneNode(true);
                const blockWrapper = newOptionBlock.querySelector('.option-block');
                const sortableList = newOptionBlock.querySelector('.sortable-list');
                const tagInput = newOptionBlock.querySelector('.tag-input');
                const removeOptionBtn = newOptionBlock.querySelector('.remove-option-btn');

                // กำหนด ID เฉพาะสำหรับ Option นี้
                const optionId = `opt_${Date.now()}`;
                blockWrapper.dataset.optionId = optionId;

                // 2.2 ทำให้ List สามารถลากและวาง (Drag-and-Drop) ได้
                new Sortable(sortableList, {
                    animation: 150,
                    ghostClass: 'sortable-ghost', // Class สำหรับเงาตอนลาก
                    draggable: '.option-value-tag', // ระบุว่า item ไหนลากได้
                    // onEnd: (evt) => {
                    //     // Logic การอัปเดตลำดับใน data model (ถ้ามี)
                    //     console.log('ลำดับใหม่:', Array.from(sortableList.children)
                    //         .filter(c => c.classList.contains('option-value-tag'))
                    //         .map(t => t.dataset.value)
                    //     );
                    // }
                });

                // 2.3 การทำงานของช่อง Input Tag (พิมพ์แล้ว Enter)
                tagInput.addEventListener('keydown', (e) => {
                    if (e.key === 'Enter' && e.target.value.trim() !== '') {
                        e.preventDefault();
                        const value = e.target.value.trim();
                        addTag(value, sortableList, tagInput);
                        e.target.value = '';
                    }
                });

                // 2.4 ปุ่มลบ Option Block
                removeOptionBtn.addEventListener('click', () => {
                    blockWrapper.remove();
                });

                optionsContainer.appendChild(newOptionBlock);
            });

            // 2.5 ฟังก์ชันสำหรับเพิ่ม Tag (Value)
            function addTag(value, listElement, inputElement) {
                const tag = document.createElement('span');
                tag.className = 'option-value-tag';
                tag.dataset.value = value;
                tag.textContent = value;
                
                const removeBtn = document.createElement('span');
                removeBtn.className = 'remove-tag-btn';
                removeBtn.innerHTML = '&times;';
                removeBtn.addEventListener('click', () => {
                    tag.remove();
                });

                tag.appendChild(removeBtn);
                listElement.insertBefore(tag, inputElement);
            }


            // --- ส่วนที่ 3 & 4: การสร้าง Variant Matrix ---
            
            generateVariantsBtn.addEventListener('click', () => {
                const options = [];
                // 3.1 อ่านข้อมูล Option และ Value ทั้งหมดจาก UI
                // **สำคัญ:** อ่านค่าตามลำดับที่แสดงผล (ที่ผ่านการลากสลับแล้ว)
                optionsContainer.querySelectorAll('.option-block').forEach(block => {
                    const optionName = block.querySelector('.option-name-input').value.trim();
                    const values = [];
                    block.querySelectorAll('.option-value-tag').forEach(tag => {
                        values.push(tag.dataset.value);
                    });

                    if (optionName && values.length > 0) {
                        options.push({ name: optionName, values: values });
                    }
                });

                if (options.length === 0) {
                    variantsTableContainer.innerHTML = `<p class="text-red-500">โปรดเพิ่มคุณสมบัติและค่าอย่างน้อย 1 รายการ</p>`;
                    variantsSection.classList.remove('hidden');
                    return;
                }

                // 3.2 สร้าง Combinations (Cartesian Product)
                const combinations = cartesianProduct(options.map(o => o.values));

                // 3.3 สร้างตาราง HTML
                let tableHtml = `<table class="min-w-full variants-table"><thead><tr>`;
                
                // สร้าง Headers (ชื่อ Option)
                options.forEach(opt => {
                    tableHtml += `<th>${opt.name}</th>`;
                });
                // สร้าง Headers (SKU, Price, Stock)
                tableHtml += `<th>SKU ย่อย</th><th>Price</th><th>Stock</th>`;
                tableHtml += `</tr></thead><tbody>`;

                // 3.4 สร้าง Rows (แต่ละ Combination)
                combinations.forEach(combo => {
                    // สร้าง SKU ย่อยอัตโนมัติ (เช่น RED-M)
                    const autoSku = combo.join('-').toUpperCase().replace(/\s+/g, '');

                    tableHtml += `<tr>`;
                    // คอลัมน์สำหรับค่า Option (เช่น "แดง", "M")
                    combo.forEach(value => {
                        tableHtml += `<td>${value}</td>`;
                    });
                    
                    // คอลัมน์สำหรับ Input
                    tableHtml += `<td><input type="text" class="variant-sku" placeholder="${autoSku}" value="${autoSku}"></td>`;
                    tableHtml += `<td><input type="number" class="variant-price" placeholder="0.00"></td>`;
                    tableHtml += `<td><input type="number" class="variant-stock" placeholder="0"></td>`;
                    tableHtml += `</tr>`;
                });

                tableHtml += `</tbody></table>`;
                variantsTableContainer.innerHTML = tableHtml;
                variantsSection.classList.remove('hidden');
            });

            // 3.5 ฟังก์ชัน Cartesian Product (หัวใจของการจับคู่)
            function cartesianProduct(arrays) {
                return arrays.reduce((a, b) => 
                    a.flatMap(x => b.map(y => [x, y].flat()))
                , [[]]);
            }


            // --- ส่วนที่ 5: Bulk Actions (Master Inputs) ---
            
            const masterPriceInput = document.getElementById('master_price');
            const applyPriceBtn = document.getElementById('btn_apply_price');
            const masterStockInput = document.getElementById('master_stock');
            const applyStockBtn = document.getElementById('btn_apply_stock');

            // 5.1 Apply Master Price
            applyPriceBtn.addEventListener('click', () => {
                const price = masterPriceInput.value;
                if (price) {
                    variantsTableContainer.querySelectorAll('.variant-price').forEach(input => {
                        input.value = price;
                    });
                }
            });

            // 5.2 Apply Master Stock
            applyStockBtn.addEventListener('click', () => {
                const stock = masterStockInput.value;
                if (stock) {
                    variantsTableContainer.querySelectorAll('.variant-stock').forEach(input => {
                        input.value = stock;
                    });
                }
            });
        });
    </script>
